#***FANATICS DATABASE***
create database fanatics;

#****FANATICS USERS****
create user fanatics@localhost identified by "fanatics";
grant all priviliges to fanatics@localhost on fanatics.*;

#*****TABLES CREATION*****

#-----USER TABLE------
create table USERS
(
    user_id int AUTO_INCREMENT,
    email_id varchar(100),
    username varchar(50),
    address varchar(200),
    contactNo varchar(20),
    user_password varchar(100),
    CONSTRAINT user_id_primary_chk PRIMARY KEY (user_id),
    CONSTRAINT email_id_unique_chk UNIQUE (email_id)
);

#=====ONE COLUMN ADDED=======
alter TABLE users
add column is_vendor int(1) default 0
after user_password;

#-----BRAND TABLE-----
create table brand
(
    brand_id int AUTO_INCREMENT PRIMARY KEY,
    brand_name varchar(100) UNIQUE,
    brand_description varchar(250),
    vendor_id int UNIQUE,
    constraint vendor_id_fk FOREIGN KEY (vendor_id) REFERENCES users(user_id)
);


#-------PRODUCT TABLE--------
create table product
(
    product_id int AUTO_INCREMENT PRIMARY KEY,
    product_name varchar(100) UNIQUE,
    product_description varchar(250),
    product_price int,
    product_size varchar(50),
    brand_id int UNIQUE,
    constraint brand_id_fk FOREIGN KEY (brand_id) REFERENCES brand(brand_id)
);
